#endif Debug.Log("[ALEX X GAMES] Mobile Performance Matrix Initialized Successfully."); } void Update() { HandleMobileTouchInputs(); } /// /// Processes physical multi-touch finger inputs optimized for mobile shooters. /// private void HandleMobileTouchInputs() { // Loop through all fingers touching the screen for (int i = 0; i < Input.touchCount; i++) { Touch touch = Input.GetTouch(i); switch (touch.phase) { case TouchPhase.Began: // Check if the touch is on the left half of the screen (Movement) if (touch.position.x < Screen.width / 2f && moveFingerId == -1) { moveFingerId = touch.fingerId; } // Check if the touch is on the right half of the screen (Looking/Aiming) else if (touch.position.x >= Screen.width / 2f && lookFingerId == -1) { lookFingerId = touch.fingerId; } break; case TouchPhase.Moved: if (touch.fingerId == moveFingerId) { moveInput = touch.deltaPosition.normalized; } else if (touch.fingerId == lookFingerId) { lookInput = touch.deltaPosition * mobileLookSensitivity; } break; case TouchPhase.Stationary: if (touch.fingerId == lookFingerId) { lookInput = Vector2.zero; } break; case TouchPhase.Ended: case TouchPhase.Clipped: if (touch.fingerId == moveFingerId) { moveFingerId = -1; moveInput = Vector2.zero; } else if (touch.fingerId == lookFingerId) { lookFingerId = -1; lookInput = Vector2.zero; } break; } } } // Public Getters so your Player Movement script can read mobile inputs seamlessly public Vector2 GetMobileMoveInput() => moveInput; public Vector2 GetMobileLookInput() => lookInput;
